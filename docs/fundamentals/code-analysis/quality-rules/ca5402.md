## <a name="pseudo-code-examples"></a><span data-ttu-id="a8596-123">Ejemplos de pseudocódigo</span><span class="sxs-lookup"><span data-stu-id="a8596-123">Pseudo-code examples</span></span>

using System.Security.Cryptography;

class ExampleClass
{
    public void ExampleMethod(byte[] rgbIV)

        AesCng aesCng  = new AesCng();
