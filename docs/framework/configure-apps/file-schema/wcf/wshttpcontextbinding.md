```xml  
                     inactivityTimeout="TimeSpan"
                     enabled="Boolean" />

      <transport clientCredentialType="Basic/Certificate/Digest/None/Ntlm/Windows"
                 proxyCredentialType="Basic/Digest/None/Ntlm/Windows"
                 realm="string"

                  maxStringContentLength="Integer" />

</wsHttpContextBinding>
```
